1. Create `bot/rate_limiter.py` (per D-11, D-12):
```python
"""Per-user command rate limiting with sliding window."""
import time
from collections import deque
from math import ceil

_user_timestamps: dict[int, deque] = {}
MAX_COMMANDS = 5
WINDOW_SECONDS = 60


def check_rate_limit(user_id: int) -> tuple[bool, int]:
    """Check if user is rate-limited.

    Returns (allowed, retry_after_seconds).
    allowed=True means the command can proceed.
    retry_after_seconds is 0 when allowed, otherwise seconds until oldest command expires.
    """
    now = time.time()
    timestamps = _user_timestamps.setdefault(user_id, deque())

    # Evict expired entries
    while timestamps and timestamps[0] < now - WINDOW_SECONDS:
        timestamps.popleft()

    if len(timestamps) >= MAX_COMMANDS:
        retry_after = ceil(WINDOW_SECONDS - (now - timestamps[0]))
        return False, max(retry_after, 1)

    timestamps.append(now)
    return True, 0
```

    from bot.rate_limiter import check_rate_limit

@patch("bot.rate_limiter.time")

        check_rate_limit(user_id=100)

    # Advance past window (61 seconds)

    allowed, retry = check_rate_limit(user_id=100)

    assert retry == 0


def test_independent_user_limits():
    from bot.rate_limiter import check_rate_limit
    # Fill user A's limit

        check_rate_limit(user_id=100)

    allowed_a, _ = check_rate_limit(user_id=100)
    assert allowed_a is False
    # User B unaffected
    allowed_b, retry_b = check_rate_limit(user_id=200)
    assert allowed_b is True
    assert retry_b == 0

    allowed, retry = check_rate_limit(user_id=100)
    assert allowed is False
    # Oldest is at t=1000, window is 60s, so expires at t=1060
    # retry_after = ceil(60 - (1030 - 1000)) = ceil(30) = 30
